/* ===================== RENDER PRODUCTS ===================== */
const allCats = { women: 'womenGrid', girls: 'childrenGrid', boys: 'childrenGrid', newborn: 'newbornGrid', sets: 'newbornGrid' };

function renderProducts(filterCatKey = 'all') {
  ['womenGrid','childrenGrid','newbornGrid'].forEach(id => {
    const el = document.getElementById(id);
    if(el) el.innerHTML = '';
  });
  const toShow = filterCatKey === 'all' ? products : products.filter(p => p.cat === filterCatKey);
  toShow.forEach(p => {
    const gridId = allCats[p.cat];
    const grid = document.getElementById(gridId);
    if(!grid) return;
    const badge = p.badge ? `<div class="card-badge ${p.badge==='New'?'new':''}">${p.badge}</div>` : '';
    const wasPriceHtml = p.wasPrice ? `<span>?${p.wasPrice.toLocaleString()}</span>` : '';
    grid.innerHTML += `
      <div class="card" data-cat="${p.cat}">
        <div class="card-img-wrap">
          <img src="${p.img}" alt="${p.name}" loading="lazy"/>
          ${badge}
          <div class="card-actions">
            <button class="quick-view" title="Wishlist">&#9825;</button>
          </div>
        </div>
        <div class="card-body">
          <div class="card-category">${p.label}</div>
          <div class="card-title">${p.name}</div>
          <div class="card-desc">${p.desc}</div>
          <div class="card-foot">
            <div class="card-price">?${p.price.toLocaleString()}${wasPriceHtml}</div>
            <button class="add-cart" onclick="addToCart(${p.id})">Add to Bag</button>
          </div>
        </div>
      </div>`;
  });

  // Hide empty section headings visually if no items
  const secIds = ['women','children','newborns'];
  secIds.forEach(sid => {
    const sec = document.getElementById(sid);
    if(sec) {
      const grid = sec.querySelector('.product-grid');
      if(grid) sec.style.display = grid.children.length === 0 ? 'none' : '';
    }
  });
  // For children section (shared grid)
  const childSec = document.getElementById('children');
  if(childSec) {
    const g = document.getElementById('childrenGrid');
    childSec.style.display = g && g.children.length ? '' : 'none';
  }
}

function filterCat(cat, btn) {
  document.querySelectorAll('.cat-pill').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  renderProducts(cat);
}

/* ===================== CART LOGIC ===================== */
let cart = {};

function addToCart(id) {
  const p = products.find(x => x.id === id);
  if(!p) return;
  if(cart[id]) { cart[id].qty++; }
  else { cart[id] = { ...p, qty: 1 }; }
  updateCartUI();
  showToast(`${p.name} added to bag`);
}

function changeQty(id, delta) {
  if(!cart[id]) return;
  cart[id].qty += delta;
  if(cart[id].qty <= 0) delete cart[id];
  updateCartUI();
}

function removeFromCart(id) {
  delete cart[id];
  updateCartUI();
}

function getSubtotal() {
  return Object.values(cart).reduce((sum, i) => sum + i.price * i.qty, 0);
}

function updateCartUI() {
  const items = Object.values(cart);
  const countEl = document.getElementById('cartCount');
  const totalQty = items.reduce((s,i) => s+i.qty, 0);
  if(totalQty > 0) {
    countEl.style.display='flex';
    countEl.textContent = totalQty;
  } else {
    countEl.style.display='none';
  }

  const drawerItems = document.getElementById('drawerItems');
  const footer = document.getElementById('drawerFooter');
  const sub = getSubtotal();
  const delivery = sub >= 1499 ? 0 : 99;
  const total = sub + delivery;

  document.getElementById('subtotalAmt').textContent = `?${sub.toLocaleString()}`;
  document.getElementById('deliveryAmt').textContent = delivery === 0 ? 'FREE' : `?${delivery}`;
  document.getElementById('totalAmt').textContent = `?${total.toLocaleString()}`;

  if(items.length === 0) {
    drawerItems.innerHTML = `<div class="empty-cart"><div class="icon">&#128722;</div><p>Your bag is empty.<br>Add something beautiful.</p></div>`;
    footer.style.display='none';
  } else {
    footer.style.display='';
    drawerItems.innerHTML = items.map(i => `
      <div class="cart-item">
        <img class="cart-item-img" src="${i.img}" alt="${i.name}"/>
        <div class="cart-item-info">
          <div class="cart-item-cat">${i.label}</div>
          <div class="cart-item-name">${i.name}</div>
          <div class="cart-item-price">?${(i.price * i.qty).toLocaleString()}</div>
          <div class="cart-item-qty">
            <button class="qty-btn" onclick="changeQty(${i.id},-1)">-</button>
            <span class="qty-num">${i.qty}</span>
            <button class="qty-btn" onclick="changeQty(${i.id},1)">+</button>
            <button class="remove-item" onclick="removeFromCart(${i.id})">Remove</button>
          </div>
        </div>
      </div>`).join('');
  }
}

/* ===================== CHECKOUT MODAL ===================== */
function openCheckout() {
  const items = Object.values(cart);
  if(items.length === 0) return;
  closeDrawer();

  const sub = getSubtotal();
  const delivery = sub >= 1499 ? 0 : 99;
  const discount = sub >= 2999 ? Math.round(sub * 0.05) : 0;
  const total = sub + delivery - discount;

  document.getElementById('billSub').textContent = `?${sub.toLocaleString()}`;
  document.getElementById('billDel').textContent = delivery === 0 ? 'FREE' : `?${delivery}`;
  document.getElementById('billDisc').textContent = `–?${discount.toLocaleString()}`;
  document.getElementById('billTotal').textContent = `?${total.toLocaleString()}`;

  document.getElementById('orderItems').innerHTML = items.map(i => `
    <div class="order-item">
      <div class="order-item-left">
        <div>
          <div class="order-item-cat">${i.label}</div>
          <div class="order-item-name">${i.name}</div>
          <div class="order-item-qty">Qty: ${i.qty}</div>
        </div>
      </div>
      <div class="order-item-price">?${(i.price*i.qty).toLocaleString()}</div>
    </div>`).join('');

  document.getElementById('checkoutModal').classList.add('open');
}

function closeCheckout() {
  document.getElementById('checkoutModal').classList.remove('open');
  openDrawer();
}
